链表的基本操作-for循环建立需要的链表 

#include<stdio.h>  
#include<stdlib.h>  
struct  node{ //链表节点
 int date;
 node*next;
};
//创建链表（关键函数）
node*creat(int Array[]){
 node *p,*pre,*head;  //pre保存当前结点的前驱结点，head为头节点
 head=new node；//创建头节点
 head->next=NULL;//头结点不需要数据域，指针域初始为NULL
 pre=head;//记录pre为head
 for(int i=0;i<5;i++){
   p=new node; //新建结点
  //将Array[i]赋给新建的结点作为数据域，也可以scanf输入
p->data=Array[i];
p->next=NULL;//新结点的指针域为NULL
pre->next=p;//前驱结点的指针域设为当前新结点的地址
pre=p;//把pre设为p，作为下个结点的前驱结点
}
return head；//返回头结点指针
}
int main（）{
int Array[5]={5,3,2,4,1};
node*L=creat（Array）；//新建链表，返回的头指针head赋给L
L=L->next;//从第一个结点开始有数据域
while（L!=NULL）{
 printf("%d",L->data);//输出每个结点的数据域
 L=L->next；
}
return 0；
}
