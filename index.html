struct node{
int x,int y;
}ssd[10];
//将ssd数组按从大到小排序
bool cmp(node a,node b){
return a.x>b.x;
}
//x相等时，按照y从大到小排序
bool cmp(node a, node b){
if (a.x!=b.x) return a.x>b.x;
else return a.y>b.y;
